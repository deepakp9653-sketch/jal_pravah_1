2. PRODUCT VISION & OBJECTIVES
2.1 Vision Statement
text

"Transform urban flood management from REACTIVE crisis response 
to PROACTIVE, data-driven prevention by giving every city ward 
a living, breathing flood intelligence system that predicts 
waterlogging before the first drop falls."
2.2 Problem Statement
text

┌──────────────────────────────────────────────────────────────────┐
│  THE PROBLEM (Quantified)                                        │
├──────────────────────────────────────────────────────────────────┤
│                                                                  │
│  CURRENT STATE:                                                  │
│  • Indian cities lose ₹15,000-50,000 Cr annually to urban       │
│    flooding (NDMA estimates)                                     │
│  • Average response time after waterlogging: 4-12 hours          │
│  • 70% of drainage infrastructure condition: UNKNOWN             │
│  • Flood prediction: based on "gut feel" + last year's memory    │
│  • Resource deployment: ad-hoc, reactive, politically driven     │
│  • Zero quantified ward-level readiness benchmarking             │
│                                                                  │
│  ROOT CAUSES:                                                    │
│  • No unified spatial view of terrain + drainage + rainfall      │
│  • Drainage capacity data exists in paper registers, if at all   │
│  • No systematic method to identify ALL vulnerable micro-zones   │
│  • Pre-monsoon preparation is checklist-based, not risk-based    │
│  • No feedback loop: same areas flood year after year            │
│                                                                  │
│  DESIRED STATE:                                                  │
│  • Every 50m × 50m patch of the city has a flood risk score      │
│  • Ward engineers know EXACTLY which spots will flood and WHY    │
│  • Resources pre-positioned HOURS before rain, not after         │
│  • Commissioner has a single dashboard: "Are we ready? Where     │
│    are we NOT ready?"                                            │
│  • Year-over-year measurable improvement in flood outcomes       │
└──────────────────────────────────────────────────────────────────┘

-- ─────────────────────────────────────
-- SPATIAL CORE TABLES

CREATE TABLE wards (
    ward_id         SERIAL PRIMARY KEY,
    ward_name       VARCHAR(100) NOT NULL,
    ward_code       VARCHAR(20) UNIQUE NOT NULL,
    zone_name       VARCHAR(100),
    area_km2        NUMERIC(10,4),
    population      INTEGER,
    geometry        GEOMETRY(MultiPolygon, 4326) NOT NULL,
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_wards_geom ON wards USING GIST(geometry);

CREATE TABLE micro_basins (
    basin_id        SERIAL PRIMARY KEY,
    basin_code      VARCHAR(20) UNIQUE NOT NULL,
    ward_id         INTEGER REFERENCES wards(ward_id),
    area_km2        NUMERIC(10,6),
    mean_elevation  NUMERIC(8,2),    -- meters
    min_elevation   NUMERIC(8,2),
    max_elevation   NUMERIC(8,2),
    mean_slope      NUMERIC(6,3),    -- degrees
    pour_point      GEOMETRY(Point, 4326),
    imperviousness  NUMERIC(5,2),    -- percentage
    curve_number    NUMERIC(5,2),
    twi             NUMERIC(8,3),    -- topographic wetness index
    upstream_area   NUMERIC(10,4),   -- km²
    fvi_score       NUMERIC(4,3),    -- 0.000 to 1.000
    fvi_class       VARCHAR(20),     -- VERY_HIGH, HIGH, MODERATE, LOW, VERY_LOW
    geometry        GEOMETRY(MultiPolygon, 4326) NOT NULL,
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_basins_geom ON micro_basins USING GIST(geometry);
CREATE INDEX idx_basins_ward ON micro_basins(ward_id);
CREATE INDEX idx_basins_fvi ON micro_basins(fvi_score DESC);

CREATE TABLE drainage_network (
    drain_id        SERIAL PRIMARY KEY,
    drain_code      VARCHAR(30),
    ward_id         INTEGER REFERENCES wards(ward_id),
    drain_type      VARCHAR(30),     -- PRIMARY, SECONDARY, TERTIARY, NALA
    material        VARCHAR(30),     -- CONCRETE, BRICK, RCC, EARTH
    shape           VARCHAR(20),     -- CIRCULAR, RECTANGULAR, TRAPEZOIDAL
    width_m         NUMERIC(6,3),
    height_m        NUMERIC(6,3),
    diameter_m      NUMERIC(6,3),    -- for circular pipes
    length_m        NUMERIC(10,2),
    slope           NUMERIC(8,6),    -- m/m
    manning_n       NUMERIC(5,4),
    capacity_m3s    NUMERIC(8,4),    -- computed full capacity
    eff_capacity    NUMERIC(8,4),    -- effective after degradation
    age_years       INTEGER,
    condition       VARCHAR(20),     -- GOOD, FAIR, POOR, CRITICAL
    last_desilt     DATE,
    geometry        GEOMETRY(MultiLineString, 4326) NOT NULL,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_drain_geom ON drainage_network USING GIST(geometry);

CREATE TABLE flood_hotspots (
    hotspot_id      SERIAL PRIMARY KEY,
    hotspot_code    VARCHAR(20) UNIQUE NOT NULL,
    basin_id        INTEGER REFERENCES micro_basins(basin_id),
    ward_id         INTEGER REFERENCES wards(ward_id),
    location_name   VARCHAR(200),
    fvi_score       NUMERIC(4,3),
    severity_class  VARCHAR(20),
    predicted_depth NUMERIC(6,2),     -- cm, for design storm
    drain_deficit   NUMERIC(5,2),     -- percentage
    pop_exposed     INTEGER,
    critical_fac    INTEGER,          -- nearby critical facilities count
    latitude        NUMERIC(10,7),
    longitude       NUMERIC(10,7),
    geometry        GEOMETRY(Point, 4326) NOT NULL,
    rank_city       INTEGER,
    rank_ward       INTEGER,
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_hotspot_geom ON flood_hotspots USING GIST(geometry);
CREATE INDEX idx_hotspot_severity ON flood_hotspots(severity_class);

CREATE TABLE rain_stations (
    station_id      SERIAL PRIMARY KEY,
    station_code    VARCHAR(20) UNIQUE,
    station_name    VARCHAR(100),
    station_type    VARCHAR(20),      -- AWS, MANUAL, RADAR
    ward_id         INTEGER REFERENCES wards(ward_id),
    latitude        NUMERIC(10,7),
    longitude       NUMERIC(10,7),
    elevation_m     NUMERIC(8,2),
    status          VARCHAR(20),      -- ACTIVE, INACTIVE, MAINTENANCE
    geometry        GEOMETRY(Point, 4326)
);

CREATE TABLE rainfall_observations (
    obs_id          BIGSERIAL PRIMARY KEY,
    station_id      INTEGER REFERENCES rain_stations(station_id),

    accumulation_24h NUMERIC(8,2)

    ward_id         INTEGER REFERENCES wards(ward_id),

    vuln_score      NUMERIC(5,2),
    response_score  NUMERIC(5,2),
